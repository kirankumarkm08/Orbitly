Edge Middleware resolves project type first:

ts
// In Edge Function / Server Component
const project = await getProject(projectId);
const niche = project.type; // 'launch' | 'ecom' | 'event' | 'real_estate' | 'static'

if (niche === 'launch' && endpoint === '/products') {
  return 404; // Products not available for launch projects
}

if (niche === 'static' && endpoint === '/events') {
  return 404; // Events not available for static projects
}
